std::string  trim_func(const std::string& str)
{
    auto find_non_space = std::find_if_not(str.begin(), str.end(), [](unsigned char f) {
        return std::isspace(f);
        });

    if (find_non_space == str.end()) return  "";

    auto last_non_space = std::find_if_not(str.rbegin(), str.rend(), [](unsigned char c) {
        return std::isspace(c);
        }).base();
    return  std::string(find_non_space, last_non_space);

}


using predicate = std::function<int(int)>;



static inline void ltrim(std::string& str, predicate const& pred = isspace)
{
    str.erase(str.begin(), std::find_if_not(str.begin(), str.end(), pred));

}

static inline void rtrim(std::string& str, predicate const& pred) {

    str.erase(std::find_if_not(str.rbegin(), str.rend(), pred).base(), str.end());


}

static inline void trim_(std::string& c, predicate const& pred = isspace)
{
    ltrim(c, pred);
    rtrim(c, pred);
}
